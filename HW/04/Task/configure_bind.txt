/etc/named.conf
  Добавляем строчки для некоей дополнительной безопасности
  (https://wiki.meurisse.org/wiki/Bind)

  options {
    
    version none;
    hostname none;
    
  }


/etc/named.conf
  Добавляем блок rate-limit

  options {

    rate-limit {
      responses-per-second 10;
      exempt-clients { 127.0.0.1; ::1; };
    };

  }


/etc/named.conf
  Слушаем не только с localhost (а со всех IP)
  То же и для IPv6
  Указываем IP, с которых допустимы рекурсивные запросы

  options {

    listen-on { any; };
    listen-on-v6 { any; };
    allow-recursion { 127.0.0.1; ::1; };

  }


sudo rndc reload
  Перезагрузка конфигурации


sudo systemctl status bind9.service
  Проверка, что всё ОК


Добавляем зону fintechtestzone.
Создаём /etc/bind/master/fintechtestzone.
  $ORIGIN fintech-admin.m1.tinkoff.cloud
  $TTL 1d ; Time a cache will keep responses

  ;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
  ;;                     Zone                                 ;;
  ;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

  @                      SOA    s-01 zonemaster01 (
                                  2018071500 ; serial
                                  12h   ; Frequency of zone transfer from slave
                                  15m   ; delay before slave retries after a zone transfer failure
                                  4w    ; Time a slave will keep the data in case it cannot contact the master
                                  1h    ; Time a cache will keep negative responses (NXDOMAIN)
                                )
  @                      NS     s-01

  s-01                   A      10.219.180.2
                         AAAA   fe80::6d:4bff:fe11:d108
                         TXT    "Any text"


Активируем файл зоны
/etc/bind/named.conf.local
  zone "fintechtestzone" {
      type master;
      file "/etc/bind/master/fintechtestzone";

      # Notify secondary servers when zone is changed
      notify yes;
  };


Перезагружаем bind
sudo rndc reconfig


Автоматическая загрузка при старте системы
systemctl enable bind
